class CoordenadaPolar {
    constructor(r, theta) {
        this.r = r;
        this.theta = theta; // √Ångulo en radianes
    }

    convertirACartesiano() {
        return new CoordenadaCartesiana(
            this.r * Math.cos(this.theta),
            this.r * Math.sin(this.theta)
        );
    }
}

class CoordenadaCartesiana {
    constructor(x, y) {
        this.x = x;
        this.y = y;
    }
}
